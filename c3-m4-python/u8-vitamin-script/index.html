# vitamin_qa.py — 完整脚本架构 # ① 导入 + 配置 # ② 定义 ask_ai() (带错误处理 + 成本追踪) # ③ 定义 process_csv() (读 CSV → 调 API → 写 CSV) # ④ 定义 generate_report() (汇总统计) # ⑤ main() 入口
来看每个部分的代码 ↓
(3)

Part 1: 导入 + 配置

# ① 导入 + 配置 import csv, time, json from datetime import datetime from openai import ( OpenAI, AuthenticationError, RateLimitError, APIConnectionError, BadRequestError, InternalServerError ) from dotenv import load_dotenv load_dotenv() client = OpenAI() # 配置 MODEL = "gpt-4o-mini" SYSTEM_PROMPT = """你是 Vitamin, 一个女性健康 AI 助手. 用温暖亲切的语气, 回复控制在 200 字以内. 涉及医疗诊断时提醒咨询医生.""" DELAY = 1.0 # 每次调用间隔 (秒) MAX_RETRIES = 3 INPUT_PRICE = 0.15 / 1_000_000 OUTPUT_PRICE = 0.60 / 1_000_000
Part 2: ask_ai() 函数 ↓
(4)

Part 2: ask_ai() -- 带全套防护

# ② 带错误处理 + 成本追踪的 ask_ai() def ask_ai(question): for attempt in range(MAX_RETRIES): try: resp = client.chat.completions.create( model=MODEL, messages=[ {"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": question} ], temperature=0.3, max_tokens=300 ) usage = resp.usage cost = usage.prompt_tokens * INPUT_PRICE + \ usage.completion_tokens * OUTPUT_PRICE return { "answer": resp.choices[0].message.content, "prompt_tokens": usage.prompt_tokens, "completion_tokens": usage.completion_tokens, "cost": cost, "status": "ok" } except AuthenticationError: return {"answer": "[AUTH ERROR]", "status": "auth_error", "prompt_tokens": 0, "completion_tokens": 0, "cost": 0} except RateLimitError: time.sleep(2 ** attempt) except (APIConnectionError, InternalServerError): time.sleep(5) except BadRequestError as e: return {"answer": f"[BAD REQUEST] {e}", "status": "bad_request", "prompt_tokens": 0, "completion_tokens": 0, "cost": 0} return {"answer": "[RETRY FAILED]", "status": "retry_failed", "prompt_tokens": 0, "completion_tokens": 0, "cost": 0}
Part 3: process_csv() ↓
(5)

Part 3: process_csv() -- 核心管道

# ③ 读 CSV → 调 API → 写 CSV def process_csv(input_file, output_file): # 读取输入 with open(input_file, "r", encoding="utf-8") as f: rows = list(csv.DictReader(f)) print(f"读取 {len(rows)} 条数据") # 逐行处理 results = [] for i, row in enumerate(rows): result = ask_ai(row["question"]) results.append({ **row, "ai_response": result["answer"], "status": result["status"], "tokens": result["prompt_tokens"] + result["completion_tokens"], "cost_usd": f"${result['cost']:.6f}" }) status = "OK" if result["status"] == "ok" else "FAIL" print(f"[{i+1}/{len(rows)}] {status} {row['question'][:25]}...") if i < len(rows) - 1: time.sleep(DELAY) # 写入输出 fieldnames = list(results[0].keys()) with open(output_file, "w", encoding="utf-8-sig", newline="") as f: writer = csv.DictWriter(f, fieldnames=fieldnames) writer.writeheader() writer.writerows(results) return results
Part 4: 生成报告 ↓
(6)

Part 4: 生成报告

# ④ 汇总统计报告 def generate_report(results): total = len(results) ok = sum(1 for r in results if r["status"] == "ok") total_tokens = sum(int(r["tokens"]) for r in results) total_cost = sum(float(r["cost_usd"].replace("$","")) for r in results) report = f""" === VITAMIN QA REPORT === Date: {datetime.now().strftime('%Y-%m-%d %H:%M')} Model: {MODEL} Total: {total} cases Success: {ok} ({ok/total*100:.0f}%) Failed: {total-ok} Tokens: {total_tokens:,} Total Cost: ${total_cost:.5f} Avg/Call: ${total_cost/total:.6f} ========================""" print(report) return report
Part 5: main() 入口 ↓
(7)

Part 5: main() 入口 -- 一键运行

# ⑤ main 入口 if __name__ == "__main__": print("=== Vitamin QA Script ===") print(f"Model: {MODEL}") print(f"Delay: {DELAY}s") print() start_time = time.time() results = process_csv("input.csv", "output.csv") elapsed = time.time() - start_time report = generate_report(results) print(f"Time: {elapsed:.1f}s") print("Results saved to output.csv") # 运行: python3 vitamin_qa.py

Vitamin 案例 -- 放上 GitHub

你的 GitHub 仓库应该长这样:

vitamin-api/ ├── README.md # 项目说明 ├── .env.example # Key 模板 (不含真实 Key) ├── .gitignore # 忽略 .env, .venv, __pycache__ ├── requirements.txt # openai, python-dotenv, tiktoken ├── input.csv # 100 条测试用例 ├── vitamin_qa.py # 主脚本 (你今天写的) └── output.csv # 结果 (可选提交)

.env.example 是什么? 它长这样:

OPENAI_API_KEY=your-key-here

给别人看格式, 但不包含真实 Key. 这是开源项目的惯例.
